PIO EQU 30H
PIC EQU 20H
DATO EQU 40H
ESTADO EQU 41H

ORG 1000H
MSJ DB "INGENIERIA E INFORMATICA"
FIN DB ?

ORG 2000H
MOV AL, ESTADO
AND AL, 07FH
OUT ESTADO, AL
MOV BX, OFFSET MSJ
MOV CL, OFFSET FIN-OFFSET MSJ
POLL: IN AL, ESTADO
      AND AL, 1
      JNZ POLL
      CMP CL, 0
      JZ FIN_L
      MOV AL, [BX]
      OUT DATO, AL
      INC BX
      DEC CL
      JMP POLL
FIN_L: INT 0
       END
