PA EQU 30H
PB EQU 31H
CA EQU 32H
CB EQU 33H

ORG 1000H
SI DB "LLAVE PRENDIDA"
FIN_SI DB ?
NO DB "LLAVE APAGADA"
FIN_NO DB ?

ORG 2000H
CLI
IN AL, PA
AND AL, 10000000B
CMP AL, 10000000B
JZ ENCENDIDA
MOV BX, OFFSET NO
MOV AL, OFFSET FIN_NO - OFFSET NO
INT 7
JMP FIN
ENCENDIDA: MOV BX, OFFSET SI
           MOV AL, OFFSET FIN_SI - OFFSET SI
           INT 7
      FIN: HLT
           END
