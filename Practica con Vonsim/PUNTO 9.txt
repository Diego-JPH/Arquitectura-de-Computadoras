MOV BX, OFFSET PASS

    JZ COMPARAR
    INT 6
    INC BX
    DEC DL
    JMP LOOP
      
COMPARAR:MOV DL, 4
         SUB BX, 4
         MOV DX, OFFSET CLAVE
LOOP_2:CMP DL, 0
       JZ SI_EXISTE
       MOV AX, [BX];ALMACENA CHAR DE LA PASS
       INC BX;PASA AL SIGUIENTE
       MOV CX, BX;TIENE LA SIGUIENTE DIRECCION DE PASS
       MOV BX, DX;PASA EL CHAR DE LA CLAVE
       MOV DX, AX;LE PONE EL CHAR ALMACENADO
       CMP [BX], DX;COMPARA
       JNZ NO_EXISTE;SI DA 0 NO SON IGUALES LAS CONTRASENIAS
       INC BX
       MOV DX, [BX];MUEVE A DX EL CHAR DE LA CLAVE
       MOV BX, CX; PASAR LA DIRECCION DE LA PASS
       DEC DL
       JMP LOOP_2 
SI_EXISTE:MOV BX, OFFSET SI
          MOV AL,OFFSET FIN_SI - OFFSET SI
          INT 7
          JMP FIN

          MOV AL, OFFSET FIN_NO - OFFSET NO
          INT 7
          JMP FIN
